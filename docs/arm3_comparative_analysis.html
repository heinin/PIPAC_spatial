metadata <- seurat_arm3_subset@meta.data %>%
  dplyr::select("Patient_ID", "DISEASESITE", "osmonths", "pfsmonths", "No_of_PIPACs") %>%
  distinct()

prop_table <- as.data.frame(table(seurat_arm3_subset@meta.data$"Celltype_Final", as.character(seurat_arm3@meta.data$Patient_ID)))
colnames(prop_table) <- c("Celltype_Final", "Patient_ID", "Freq")
prop_table <- spread(prop_table, Celltype_Final, Freq)
# Converting to percentage
prop_table[,2:length(prop_table)] <- (prop_table[,2:length(prop_table)]/rowSums(prop_table[,2:length(prop_table)]))*100
prop_table <- gather(prop_table, Celltype_Final, Freq, names(prop_table)[2:length(names(prop_table))], factor_key=TRUE)

# Adding survival data
prop_survival <- merge(prop_table, metadata, by = "Patient_ID")
prop_survival$osmonths <- unlist(prop_survival$osmonths)
prop_survival$pfmonths <- unlist(prop_survival$pfmonths)

# Plotting
plot_list <- lapply(sort(unique(seurat_arm3_subset$"Celltype_Final")), function(ct){
  prop_survival %>%
    filter("Celltype_Final" == ct) %>%
    ggplot(aes(x = Freq, y = osmonths, color = "Celltype_Final")) +
    geom_point() +
    scale_color_manual(name = "Celltype_Final", values = pipac_celltype_col) +

                 rr.digits = 3, p.digits = 3, small.p = T,

    ylab("Overall Survival") +

    theme_classic() +
